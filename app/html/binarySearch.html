function binarySearch(items, target, start, end) {
	
	var midpoint = (start + end) / 2;
	
	// If the array is too small, return null, we did not find the target.
	if end < start { return null; }
	
	// If the value at index midpoint is the same as the target, return the index.
	if (items[midpoint] == target) { return midpoint; }
	
	// Otherwise, figure out which half is relevant and recurse.
	if (items[midpoint] <  target) {
		return binarySearch(items, target, midpoint + 1, end);
	}
	
	else {
		return binarySearch(items, target, start, midpoint - 1);		
	}
}
